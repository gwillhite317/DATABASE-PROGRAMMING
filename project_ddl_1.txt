DROP TABLE IF EXISTS talksTo CASCADE;
DROP TABLE IF EXISTS books CASCADE;
DROP TABLE IF EXISTS offers CASCADE;
DROP TABLE IF EXISTS manages CASCADE;
DROP TABLE IF EXISTS pays CASCADE;
DROP TABLE IF EXISTS takes CASCADE;
DROP TABLE IF EXISTS prescribes CASCADE;
DROP TABLE IF EXISTS worksIn CASCADE;
DROP TABLE IF EXISTS has CASCADE;
DROP TABLE IF EXISTS bill CASCADE;
DROP TABLE IF EXISTS secretary CASCADE;
DROP TABLE IF EXISTS unit CASCADE;
DROP TABLE IF EXISTS treatment CASCADE;
DROP TABLE IF EXISTS appointment CASCADE;
DROP TABLE IF EXISTS patient CASCADE;
DROP TABLE IF EXISTS doctor CASCADE;

CREATE TABLE doctor (
    dID VARCHAR(20) PRIMARY KEY,
    dLastName VARCHAR(10),
    dFirstName VARCHAR(10),
    dType VARCHAR(15)	
);

CREATE TABLE patient (
    pID VARCHAR(10) PRIMARY KEY,
    pFirstName VARCHAR(10),
    pLastName VARCHAR(10),
    pAdr VARCHAR(30),
    sID VARCHAR(3)	
);

CREATE TABLE appointment (
    aID VARCHAR(10) PRIMARY KEY,
    aDate VARCHAR(10),
    aTime VARCHAR(10),
    pId VARCHAR(10),
    FOREIGN KEY (pID) REFERENCES patient(pID)
);

CREATE TABLE treatment (
    tID VARCHAR(10) PRIMARY KEY,
    tName VARCHAR(20)
);

CREATE TABLE unit (
    uID VARCHAR(10) PRIMARY KEY,
    uName VARCHAR(10),
    uPhone VARCHAR(10)
);

CREATE TABLE secretary (
    sID VARCHAR(3) PRIMARY KEY,
    sName VARCHAR(20)
);

CREATE TABLE bill (
    bID VARCHAR(10) PRIMARY KEY,
    bAmount VARCHAR(10),
    bDate VARCHAR(10)
);

CREATE TABLE has (
    dID VARCHAR(10),
    pID VARCHAR(10),
    PRIMARY KEY (dID, pID)
);

CREATE TABLE worksIn (
    dID VARCHAR(10),
    uID VARCHAR(10),
    PRIMARY KEY (dID, uID)
);

CREATE TABLE pays (
    bId VARCHAR(10) PRIMARY KEY,
    bAmount VARCHAR(10),
    bDate VARCHAR(10),
    pID VARCHAR(10),
    FOREIGN KEY (pID) REFERENCES patient(pID)
);

CREATE TABLE takes (
    tID VARCHAR(10),
    pID VARCHAR(10),
    PRIMARY KEY (tID, pID)
);

CREATE TABLE prescribes (
    tID VARCHAR(10),
    dID VARCHAR(10),
    PRIMARY KEY (tID, dID)
);

CREATE TABLE manages (
    bID VARCHAR(10) PRIMARY KEY,
    bAmount VARCHAR(10),
    bDate VARCHAR(10),
    sID VARCHAR(3),
    FOREIGN KEY (sID) REFERENCES secretary(sID)
);

CREATE TABLE offers (
    aID VARCHAR(10) PRIMARY KEY,
    aDate VARCHAR(10),
    aTime VARCHAR(10),
    dID VARCHAR(10),
    FOREIGN KEY (dID) REFERENCES doctor(dID)
);

CREATE TABLE books (
    pID VARCHAR(10),
    aID VARCHAR(10),
    PRIMARY KEY (pID, aID)
);

CREATE TABLE talksTo (
    sID VARCHAR(3),
    pId VARCHAR(10),
    PRIMARY KEY (sID, pID)
);
